package service

import entity.Card
import entity.Game
import entity.Player

class GameService {
    var currentGame = Game(
        listOf(
            Player("", arrayListOf<Card?>(null, null, null)),
            Player("", arrayListOf<Card?>(null, null, null))
        )
    )

    fun handOutCards() {
        for (i in 0..currentGame.players.size - 1) {
            currentGame.players[i].hand =
                arrayListOf(
                    currentGame.drawPile.draw(),
                    currentGame.drawPile.draw(),
                    currentGame.drawPile.draw()
                )
        }
    }

    fun resetPlacedCards() {
        currentGame.placedCards = arrayOf(
            currentGame.drawPile.draw(),
            currentGame.drawPile.draw(),
            currentGame.drawPile.draw()
        )
    }
}
